#include <include/libssh/sftp.h>

struct sftp_attributes_struct {
    // fields

    ssh_string acl;
    uint32_t atime;
    uint64_t atime64;
    uint32_t atime_nseconds;
    uint64_t createtime;
    uint32_t createtime_nseconds;
    uint32_t extended_count;
    ssh_string extended_data;
    ssh_string extended_type;
    uint32_t flags;
    uint32_t gid;
    char* group;
    char* longname;
    uint32_t mtime;
    uint64_t mtime64;
    uint32_t mtime_nseconds;
    char* name;
    char* owner;
    uint32_t permissions;
    uint64_t size;
    uint8_t type;
    uint32_t uid;
};
