enum ssh_scp_request_types

Overview

#include <libssh.h>

enum ssh_scp_request_types {
    SSH_SCP_REQUEST_NEWDIR  =1,
    SSH_SCP_REQUEST_NEWFILE,
    SSH_SCP_REQUEST_EOF,
    SSH_SCP_REQUEST_ENDDIR,
    SSH_SCP_REQUEST_WARNING,
};
